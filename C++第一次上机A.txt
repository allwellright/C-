#include<iostream>
#include<algorithm>
using namespace std;
class Number
{
	int num;
	public:
		Number(int a)
		{
			num = a;
		}
		Number()
		{
		}
	Number maxDivisor(Number b)
	{
		Number temp;
		int i;
		if(num>b.num)
		i = b.num;
		else i=num;
		for(;i>=1;i--)
		{
			if(num%i==0&&b.num%i==0)
			{
				temp.num = i;
				break;
			}
		}
		return temp;
	}
	void show()
	{
		cout<<num<<endl;
	}
 }; 
int main() {

    int n, n1, n2;

    std::cin >> n;

    for(int i = 0; i < n; ++i) {

        std::cin >> n1 >> n2;

        Number no1(n1), no2(n2);

        Number no3 = no1.maxDivisor(no2);  //最大公约数

        no3.show();

    }
    return 0;

}
