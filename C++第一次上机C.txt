#include <iostream>
#include <algorithm>
using namespace std;
class Number
{
	public:
	int num;
	int bit;
	Number(int a,int b)
	{
		num =a;
		bit = b;
	}
	Number()
	{
		
	} 
	void show()
	{
		cout<<num<<endl;
	}
	Number converto(int n2)
	{
		Number temp;
		temp.bit = n2;
		int t = num;
		int res=0;			
		while(t!=0)
		{
			res += t%n2;
			res = res*10;
			t = t/n2; 
		}
		res = res/10;
		temp.num = res;
		return temp;
	}
};
int main() {

    int n, n1, n2;

    std::cin >> n;

    for(int i = 0; i < n; ++i) {

        std::cin >> n1 >> n2;

        Number no1(n1, 10);     //n1是10进制正整数

        Number no3 = no1.converto(n2);  //no3是n2进制的正整数

        no3.show();     //输出结果

    }

}
